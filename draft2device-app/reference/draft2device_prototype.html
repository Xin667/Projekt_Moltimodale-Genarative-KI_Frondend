Schritt 5 · Quellcode (optional)

Aus dem Zustandsautomaten kompiliert

Der Code wird aus der Zustandsmaschine erzeugt, nicht frei generiert. Parameter rechts ändern die markierten Werte live — ohne Neugenerierung.

// Draft2Device · generiert aus Zustandsautomat v4
// Ziel: Seeed XIAO ESP32-C3 · Grove
#include <Grove_Moisture.h>

const int SCHWELLE_DURST   = 30;  // % Feuchte
const int JAMMER_INTERVALL = 30;  // Sekunden
const int BLINK_ANZAHL     = 5;   // grünes Blinken

enum Zustand { NEUTRAL, DURST, GIESSEN, UEBERFLUTUNG };
Zustand aktuell = NEUTRAL;

void loop() {
  int feuchte = leseFeuchtigkeit();
  switch (aktuell) {
    case NEUTRAL:
      if (feuchte < SCHWELLE_DURST) aktuell = DURST;
      break;
    case DURST:
      ledRot();
      jammernAlle(JAMMER_INTERVALL);
      if (anstiegErkannt()) aktuell = GIESSEN;
      break;
    case GIESSEN:
      blinkGruen(BLINK_ANZAHL);
      froehlicherSound();
      aktuell = NEUTRAL;
      break;
  }
}
